**useCountUp.ts** (NEW — `src/hooks/useCountUp.ts`):
```typescript
import { useMotionValue, useSpring, useMotionValueEvent } from 'motion/react'
import { useState, useEffect } from 'react'
import { gentle } from '../motion/transitions'

export function useCountUp(target: number): number {
  const motionValue = useMotionValue(0)
  const spring = useSpring(motionValue, gentle)
  const [display, setDisplay] = useState(0)

  useEffect(() => {
    motionValue.set(target)
  }, [target, motionValue])

  useMotionValueEvent(spring, 'change', (latest) => {
    setDisplay(Math.round(latest))
  })

  return display
}
```

Per locked decision: uses `gentle` spring (~400ms settling), returns integer, driven by `useMotionValue` + `useSpring` + `useMotionValueEvent`.

**Skeleton.tsx** (NEW — `src/components/ui/Skeleton.tsx`):
```typescript
type SkeletonVariant = 'text' | 'circle' | 'card' | 'block'

interface SkeletonProps {
  variant?: SkeletonVariant
  width?: number | string
  height?: number | string
  className?: string
}

export default function Skeleton({
  variant = 'block',
  width,
  height,
  className = '',
}: SkeletonProps) {
  const shapeClass = {
    text: 'rounded-full',
    circle: 'rounded-full',
    card: 'rounded-[var(--radius-lg)]',
    block: 'rounded-[var(--radius-md)]',
  }[variant]

  return (
    <div
      className={`skeleton-shimmer ${shapeClass} ${className}`}
      style={{ width, height }}
      aria-hidden="true"
    />
  )
}
```

Per locked decisions:
- Shape variants: text, circle, card, block
- CSS shimmer (not Motion) for repeating loop
- aria-hidden for accessibility
- No `bg-*` Tailwind class — the `.skeleton-shimmer` CSS class handles all background (per Pitfall 4)
