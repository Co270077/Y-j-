```typescript
import * as m from 'motion/react-m'
import { scaleIn, tap } from '../../motion/variants'
import { hapticLight } from '../../utils/haptics'

interface FABProps {
  onClick: () => void
  label: string
}

export default function FAB({ onClick, label }: FABProps) {
  return (
    <m.button
      variants={scaleIn}
      initial="initial"
      animate="animate"
      whileTap={tap.whileTap}
      onClick={() => { hapticLight(); onClick() }}
      aria-label={label}
      className="fixed bottom-[calc(4rem+16px+env(safe-area-inset-bottom,0px))] right-4 z-40 w-14 h-14 rounded-full bg-bamboo text-warm-white shadow-lg shadow-black/30 flex items-center justify-center cursor-pointer"
    >
      <svg width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2.5" strokeLinecap="round">
        <line x1="12" y1="5" x2="12" y2="19" />
        <line x1="5" y1="12" x2="19" y2="12" />
      </svg>
    </m.button>
  )
}
```

Note on `tap` variant usage: The `tap` variant uses `whileTap` key which is a prop, not a variant state. Access it as `tap.whileTap` and spread or assign directly to the `whileTap` prop. If TypeScript complains about the type, cast or extract the value: `whileTap={{ scale: 0.97 }}` with snappy transition inline.

FAB positioning: `bottom: calc(4rem + 16px + env(safe-area-inset-bottom, 0px))` — 4rem is the h-16 bottom nav, 16px gap, plus safe area. The `0px` fallback prevents calc failure on non-notched devices.

Modify `src/components/schedule/Timeline.tsx`:

1. **Import useLocation:** `import { useLocation } from 'react-router-dom'`

2. **Add location and scroll target ref:**
   ```typescript
   const location = useLocation()
   const scrollTargetRef = useRef<HTMLDivElement>(null)
   ```

3. **Replace the current auto-scroll logic.** The existing effect only fires on `day` change and targets `block: 'center'`. Replace with:
   ```typescript
   useEffect(() => {
     const timer = setTimeout(() => {
       scrollTargetRef.current?.scrollIntoView({ behavior: 'smooth', block: 'start' })
     }, 80)  // 80ms delay so route transition settles before scroll
     return () => clearTimeout(timer)
   }, [location.pathname, day])
   ```
   The 80ms delay prevents scroll from fighting the page entrance animation (gentle spring takes ~400ms to settle, but scroll feels natural starting at 80ms).

4. **Determine scroll target task.** After `sortedTasks` is computed and `currentMinutes` is set, find the scroll target:
   ```typescript
   const scrollTargetTask = sortedTasks.find(task => {
     const start = minutesSinceMidnight(task.startTime)
     const end = start + task.durationMinutes
     return start <= currentMinutes && end > currentMinutes  // current task
   }) ?? sortedTasks.find(task => {
     return minutesSinceMidnight(task.startTime) > currentMinutes  // next upcoming
   }) ?? sortedTasks[sortedTasks.length - 1]  // late night fallback: last task
   ```

5. **Assign scrollTargetRef** to the scroll target task's wrapper div. Replace the existing `ref={isCurrentTask ? currentTimeRef : undefined}` with:
   ```typescript
   ref={task === scrollTargetTask ? scrollTargetRef : undefined}
   ```
   Remove the old `currentTimeRef` — it's fully replaced by `scrollTargetRef` with better targeting logic.

6. **Add bamboo left-border accent** on the scroll target task (current or next upcoming). On the wrapper div that gets the ref, add a conditional class:
   ```typescript
   <div
     key={task.id}
     ref={task === scrollTargetTask ? scrollTargetRef : undefined}
     className={task === scrollTargetTask ? 'border-l-[3px] border-bamboo pl-2 -ml-2 rounded-sm' : ''}
   >
   ```
   The 3px left border with bamboo color visually anchors the current/next time block. The `-ml-2 pl-2` compensates for the border so content doesn't shift.

7. **Keep the existing "NOW" indicator** — it shows for `isCurrentTask` (the task whose time range includes now). The scroll target may be a different task (next upcoming when between blocks). Both can coexist: NOW indicator shows on the current task, left-border accent shows on the scroll target.
